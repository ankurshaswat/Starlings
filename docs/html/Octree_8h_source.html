105  N& getCell(const float pos[3], Callback* callback = NULL)
106  {
107  Point ppos(pos);
108  // std::cout<<"MIN-"<<_min.x<<" "<<_min.y<<" "<<_min.z<<std::endl;
109  // std::cout<<"MAX-"<<_max.x<<" "<<_max.y<<" "<<_max.z<<std::endl;
110 
111  assert(ppos >= _min && ppos < _max);
112  Point currMin(_min);
113  Point currMax(_max);
114  Point delta = _max - _min;
115  if (!_root)
116  _root = new OctreeNode();
117  OctreeNode* currNode = _root;
118  while (delta >= _cellSize)
119  {
120  bool shouldContinue = true;
121  if (callback)
122  shouldContinue = callback->operator()(currMin, currMax, currNode->_nodeData);
123  if (!shouldContinue)
124  break;
125  Point mid = (delta * 0.5f) + currMin;
126  Point newMin(currMin);
127  Point newMax(currMax);
128  int index = 0;
129  COMPUTE_SIDE(index, 1, ppos.x, mid.x, newMin.x, newMax.x)
130  COMPUTE_SIDE(index, 2, ppos.y, mid.y, newMin.y, newMax.y)
131  COMPUTE_SIDE(index, 4, ppos.z, mid.z, newMin.z, newMax.z)
132  if (!(currNode->_children[index]))
133  currNode->_children[index] = new OctreeNode();
134  currNode = currNode->_children[index];
135  currMin = newMin;
136  currMax = newMax;
137  delta = currMax - currMin;
138  }
139  return currNode->_nodeData;
140  }
